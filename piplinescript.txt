pipeline {
    agent any
    stages {
        stage('Initialization') {
            environment {
                   Hash_ID = sh (returnStdout: true, script: 'git log --format="%H" -n 1')
		   Branch = sh (returnStdout: true, script: 'git branch | grep -e "^*" | cut -f 2' )
                  }
            steps {
                sh 'echo Hash_ID = ${GIT_COMMIT}'
		sh 'echo Branch = $Branch'
		sh 'env'

		echo env.GIT_COMMIT
                }
        }
    }
}
