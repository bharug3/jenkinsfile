pipeline {
    agent any
    tools {
        git 'git'
    }
    stages {
        stage('Initialization') {
            environment { 
                   Hash_ID = sh (returnstdout: true, script: "git log -1").trim()
                   Branch_NAME = sh (returnstdout: true, script: "git branch").trim()
            }
            steps {
                sh 'echo $Hash_ID'
                sh 'echo $Branch_NAME'
            }
        }
    }
}
