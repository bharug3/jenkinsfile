pipeline {
    agent any
    stages {
        stage('Initialization') {
            environment {
                   Hash_ID = sh (returnStdout: true, script: 'git log --format="%H" -n 1')
		   Branch = sh (returnStdout: true, script: 'git rev-parse --abbrev-ref HEAD' )
                  }
            steps {
                sh 'echo $Hash_ID'
		sh 'echo $Branch'
                }
        }
    }
}
